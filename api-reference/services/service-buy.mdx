---
title: "Purchase Service"
api: "POST /service-buy"
description: "Endpoint to buy a service using blockchain transaction"
---

### Body

<ParamField body="transactionHash" type="string" required>
  The hash of the blockchain transaction
</ParamField>

<ParamField body="reference" type="string" required>
  Reference number for the service
</ParamField>

<ParamField body="productId" type="string" required>
  ID of the product being purchased
</ParamField>

<ParamField body="amount" type="number" required>
  Amount of the transaction
</ParamField>

<ParamField body="carrierId" type="number" required>
  ID of the carrier
</ParamField>

<ParamField body="chain" type="string" required>
  The blockchain network used for the transaction
</ParamField>

<ParamField body="coin" type="string" required>
  The cryptocurrency used for the transaction
</ParamField>

<ParamField body="walletAddress" type="string" required>
  The wallet address used for the transaction
</ParamField>

### Response

<ResponseField name="id" type="string">
  Unique identifier for the transaction
</ResponseField>

<ResponseField name="transactionHash" type="string">
  The hash of the blockchain transaction
</ResponseField>

<ResponseField name="serviceTransactionId" type="string">
  ID of the service transaction
</ResponseField>

<ResponseField name="folio" type="string">
  Folio number for the transaction
</ResponseField>

<ResponseField name="reference" type="string">
  Reference number for the service
</ResponseField>

<ResponseField name="carrierName" type="string">
  Name of the carrier
</ResponseField>

<ResponseField name="carrierId" type="number">
  ID of the carrier
</ResponseField>

<ResponseField name="productId" type="string">
  ID of the product purchased
</ResponseField>

<ResponseField name="amount" type="number">
  Amount of the transaction
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp of when the transaction was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp of when the transaction was last updated
</ResponseField>
